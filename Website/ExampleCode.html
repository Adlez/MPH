One example of some code I've written in Visual Studio is the attack button for a program I made for testing
character levels, armour and skills and everything else for my tabletop rpg. The objective is to use the program to find what level and
kind of enemies the players could handle.
Private Sub btnAttack_Click_1(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles btnAttack.Click
'The following will happen when the user clicks the "attack" button
'First the program will find out what weapons the user has equipped and in which hand
'Then it calculates the user's "Accuracy", their chance to hit
'Next it finds the amount of damage that is to be dealt out.
radSkillSelectNoneDerek.Checked = True
radSkillSelectNoneVex.Checked = True



TurnCounter = TurnCounter + 1
lblHitAimTarget.Text = "Hit"
'If there is no weapon equipped
If EmptyHands = True And MeleeTwoHanded = False And MeleeLeft = False And RangeLeft = False And RangeRight = False And RangedTwoHaned = False And MeleeRight = False Then
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = Strength
'If a two handed melee weapon is equipped
ElseIf MeleeTwoHanded = True And MeleeLeft = False And RangeLeft = False And RangeRight = False And RangedTwoHaned = False And MeleeRight = False Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
'If a melee weapon is in the left hand and not the right
ElseIf MeleeLeft = True And MeleeTwoHanded = False And RangeLeft = False And RangedTwoHaned = False And MeleeRight = False Then
If LeftHandIsDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
ElseIf RightHandisDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = LeftOutputWeaponDamage
End If
'If a melee weapon is equipped in the Right hand and not the left
ElseIf MeleeRight = True And MeleeTwoHanded = False And RangeLeft = False And RangedTwoHaned = False And MeleeLeft = False Then
If RightHandisDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = RightOutputWeaponDamage + RightWeaponNTValue
ElseIf LeftHandIsDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = RightOutputWeaponDamage
End If
'If dual-wielding two melee weapons
ElseIf MeleeRight = True And MeleeLeft = True And MeleeTwoHanded = False And RangeLeft = False And RangeRight = False And RangedTwoHaned = False Then
If LeftHandIsDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = RightOutputWeaponDamage + LeftOutputWeaponDamage + LeftWeaponNTValue
ElseIf RightHandisDom = True Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
txtDamage.Text = RightOutputWeaponDamage + LeftOutputWeaponDamage + RightWeaponNTValue
End If
'If wielding a two-handed ranged weapon
ElseIf RangedTwoHaned = True And RangeRight = False And RangeLeft = False And MeleeLeft = False And MeleeRight = False And MeleeTwoHanded = False Then
lblHitAimTarget.Text = "Aim"
txtHitAimTarget.Text = Accuracy + Dexterity + LeftOutputWeaponAim
txtDamage.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
'If wielding a ranged weapon in the left hand and not the right
ElseIf RangeLeft = True And RangeRight = False And RangedTwoHaned = False And MeleeLeft = False And MeleeRight = False And MeleeTwoHanded = False
lblHitAimTarget.Text = "Aim"
txtHitAimTarget.Text = Accuracy + Dexterity + LeftOutputWeaponAim
txtDamage.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
ElseIf RightHandisDom = True Then
lblHitAimTarget.Text = "Aim"
txtHitAimTarget.Text = Accuracy + LeftOutputWeaponAim
txtDamage.Text = LeftOutputWeaponDamage
End If
'If wielding a ranged weapon in the right hand and not the left
ElseIf RangeRight = True And RangedTwoHaned = False And RangeLeft = False And MeleeLeft = False And MeleeRight = False And MeleeTwoHanded = False Then
If RightHandisDom = True Then
lblHitAimTarget.Text = "Aim"
txtHitAimTarget.Text = Accuracy + Dexterity + RightOutputWeaponAim
txtDamage.Text = RightOutputWeaponDamage + RightWeaponNTValue
ElseIf LeftHandIsDom = True Then
lblHitAimTarget.Text = "Aim"
txtHitAimTarget.Text = Accuracy + RightOutputWeaponAim
txtDamage.Text = RightOutputWeaponDamage
End If
'If dual wielding a ranged weapon and a melee weapon
Else
txtHitAimTarget.Text = Dexterity + LeftOutputWeaponAim + RightOutputWeaponAim
If RangeRight = True And MeleeLeft = True And RangeLeft And MeleeRight = False And RangedTwoHaned = False And MeleeTwoHanded = False Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
If RightHandisDom = True Then
lblHitAimTarget.Text = "Hit"
txtRangeDamageDealt.Text = RightOutputWeaponDamage + RightWeaponNTValue
txtDamage.Text = LeftOutputWeaponDamage
txtHitAimTarget.Text = txtHitAimTarget.Text + Accuracy
ElseIf LeftHandIsDom = True Then
lblHitAimTarget.Text = "Hit"
txtRangeDamageDealt.Text = RightOutputWeaponDamage
txtDamage.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
End If

ElseIf RangeLeft = True And MeleeRight = True And RangeRight And MeleeLeft = False And MeleeTwoHanded = False Then
lblHitAimTarget.Text = "Hit"
Dim HITValue As Random
Dim HIT As Integer
HITValue = New Random
HIT = HITValue.Next(1, 11)
txtHitAimTarget.Text = HIT + Dexterity
If RightHandisDom = True Then
lblHitAimTarget.Text = "Hit"
txtRangeDamageDealt.Text = LeftOutputWeaponDamage
txtDamage.Text = RightOutputWeaponDamage + RightWeaponNTValue
ElseIf LeftHandIsDom = True Then
lblHitAimTarget.Text = "Hit"
txtRangeDamageDealt.Text = LeftOutputWeaponDamage + LeftWeaponNTValue
txtDamage.Text = RightOutputWeaponDamage
txtHitAimTarget.Text = txtHitAimTarget.Text + Accuracy
End If
End If
End If

'specific character traits
If radCharacterDerek.Checked = True Then

If RangeLeft = False And RangeRight = False And RangedTwoHaned = False Then

End If

ElseIf radCharacterVex.Checked = True Then
If RangedTwoHaned Or RangeRight Or RangeLeft = True Then

End If
ElseIf radCharacterChester.Checked = True Then

End If

End If

End If

End If

End If
